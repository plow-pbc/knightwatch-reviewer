+# Mount a Plow DMG and assert it has no custom volume icon (PLO-29 regression
+# guard) and that the volume name is "Plow". Detaches before exiting.
+set -euo pipefail
+
+usage() {
+  cat <<'EOF'
+Usage: plowd-verify-dmg [--dmg <path>]
+
+Mounts the given DMG, asserts the mounted volume is named "Plow" and has no
+.VolumeIcon.icns at its root, then detaches. Defaults to .build/Plow.dmg
+relative to the repo root.
+EOF
+}
+
+repo_root="$(cd "$(dirname "${BASH_SOURCE[0]}")/../../.." && pwd)"
+dmg_path="${repo_root}/.build/Plow.dmg"
+
+while [[ $# -gt 0 ]]; do
+  case "$1" in
+    --dmg)
+      dmg_path="${2:-}"
+      shift 2
+      ;;
+    -h|--help)
+      usage
+      exit 0
+      ;;
+    *)
+      echo "error: unknown argument: $1" >&2
+      usage >&2
+      exit 1
+      ;;
+  esac
+done
+
+if [[ "${dmg_path}" != /* ]]; then
+  dmg_path="${repo_root}/${dmg_path}"
+fi
+
+[[ -f "${dmg_path}" ]] || { echo "error: DMG not found: ${dmg_path}" >&2; exit 1; }
+
+# Refuse to run when a Plow volume is already mounted: the new mountpoint
+# would be "Plow 1" / "Plow 2" / ..., which would mask a real regression in
+# the volume-name assertion below.
+mounted_plow_volumes="$(hdiutil info 2>/dev/null | grep -Eo '/Volumes/Plow( [0-9]+)?$' || true)"
+mounted_plow_volumes="$(printf '%s\n' "${mounted_plow_volumes}" | sort -u)"
+if [[ -n "${mounted_plow_volumes}" ]]; then
+  echo "error: a Plow volume is already mounted; eject it before verifying:" >&2
+  printf '%s\n' "${mounted_plow_volumes}" >&2
+  exit 1
+fi
+
+mounted_volume=""
+cleanup() {
+  local rc=$?
+  if [[ -n "${mounted_volume}" ]]; then
+    if [[ "${rc}" -eq 0 ]]; then
+      # Success path: detach failure (e.g. volume busy) must be fatal.
+      # Otherwise the next release retry trips ensure_no_mounted_plow_volume

+      fi

+  fi

   592	# ---------------------------------------------------------------------------
   593	# Cross-platform verifier tests. These drive plowd-verify-dmg through a fake
   594	# `hdiutil` that points the verifier at a tmp directory whose contents and
   595	# basename the test controls. They exercise the same accept/reject code path
   596	# the real release run takes — no env-skip, no green skip — so the verifier's
   597	# decision logic is covered on every CI lane (including non-macOS ones that
   598	# can't run real hdiutil at all).
   599	# ---------------------------------------------------------------------------
   600	def _write_verifier_fake_hdiutil(fake_bin: Path, mount_path: Path) -> None:
   601	    """Stub `hdiutil` so plowd-verify-dmg attaches to ``mount_path``.
   602	
   603	    The verifier shells out to `hdiutil info` (mount listing), `hdiutil
   604	    attach` (which prints a tabular line whose final column is the mount
   605	    point) and `hdiutil detach` on cleanup. The stub:
   606	
   607	    - prints nothing for `info` (no /Volumes/Plow already mounted)
   608	    - prints `/dev/disk9s1 Apple_APFS <mount_path>` for `attach`
   609	    - succeeds for `detach`
   610	
   611	    The test pre-stages ``mount_path`` with whatever contents and basename
   612	    it wants the verifier to see.
   613	    """
   614	    fake_bin.mkdir(exist_ok=True)
   615	    hdiutil = fake_bin / "hdiutil"
   616	    hdiutil.write_text(
   617	        "#!/usr/bin/env bash\n"
   618	        "case \"$1\" in\n"
   619	        "  info) exit 0 ;;\n"
   620	        f"  attach) printf '/dev/disk9s1 Apple_APFS %s\\n' '{mount_path}'; exit 0 ;;\n"
   621	        "  detach) exit 0 ;;\n"
   622	        "  *) exit 1 ;;\n"
   623	        "esac\n"
   624	    )
   625	    hdiutil.chmod(0o755)
