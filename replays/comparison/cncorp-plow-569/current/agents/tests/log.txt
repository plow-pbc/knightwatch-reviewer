+#!/usr/bin/env bash
+# Mount a Plow DMG and assert it has no custom volume icon (PLO-29 regression
+# guard) and that the volume name is "Plow". Detaches before exiting.
+set -euo pipefail
+
+usage() {
+  cat <<'EOF'
+Usage: plowd-verify-dmg [--dmg <path>]
+
+Mounts the given DMG, asserts the mounted volume is named "Plow" and has no
+.VolumeIcon.icns at its root, then detaches. Defaults to .build/Plow.dmg
+relative to the repo root.
+EOF
+}
+
+repo_root="$(cd "$(dirname "${BASH_SOURCE[0]}")/../../.." && pwd)"
+dmg_path="${repo_root}/.build/Plow.dmg"
+
+while [[ $# -gt 0 ]]; do
+  case "$1" in
+    --dmg)
+      dmg_path="${2:-}"
+      shift 2
+      ;;
+    -h|--help)
+      usage
+      exit 0
+      ;;
+    *)
+      echo "error: unknown argument: $1" >&2
+      usage >&2
+      exit 1
+      ;;
+  esac
+done
+
+if [[ "${dmg_path}" != /* ]]; then
+  dmg_path="${repo_root}/${dmg_path}"
+fi
+
+[[ -f "${dmg_path}" ]] || { echo "error: DMG not found: ${dmg_path}" >&2; exit 1; }
+
+# Refuse to run when a Plow volume is already mounted: the new mountpoint
+# would be "Plow 1" / "Plow 2" / ..., which would mask a real regression in
+# the volume-name assertion below.
+mounted_plow_volumes="$(hdiutil info 2>/dev/null | grep -Eo '/Volumes/Plow( [0-9]+)?$' || true)"
+mounted_plow_volumes="$(printf '%s\n' "${mounted_plow_volumes}" | sort -u)"
+if [[ -n "${mounted_plow_volumes}" ]]; then
+  echo "error: a Plow volume is already mounted; eject it before verifying:" >&2
+  printf '%s\n' "${mounted_plow_volumes}" >&2
+  exit 1
+fi
+
+mounted_volume=""
+cleanup() {
+  local rc=$?
+  if [[ -n "${mounted_volume}" ]]; then
+    if [[ "${rc}" -eq 0 ]]; then
+      # Success path: detach failure (e.g. volume busy) must be fatal.
+      # Otherwise the next release retry trips ensure_no_mounted_plow_volume
+      # in plowd-beta-package-signed and aborts on a leaked mount.
+      if ! hdiutil detach "${mounted_volume}"; then
+        echo "error: failed to detach ${mounted_volume} after successful verify" >&2
+        exit 1
+      fi
+    else
+      # Failure path: don't let detach noise mask the real error.
+      hdiutil detach "${mounted_volume}" >/dev/null 2>&1 || true
+    fi
+  fi
+  exit "${rc}"
+}

+

def _provision_fake_bin(fake_bin: Path, systemctl_log: Path) -> None:
    """Write stub commands sufficient to run provision-runtime.sh on macOS."""
    stubs = {
        # zstd just needs to be on PATH so the apt-get install branch is skipped
        "zstd": "#!/bin/sh\n",
        # sha256sum: hash stdin or file args, mimicking GNU coreutils behaviour
        "sha256sum": textwrap.dedent("""\
            #!/usr/bin/env python3
            import hashlib, sys
            from pathlib import Path
            args = [a for a in sys.argv[1:] if a != "--"]
            if args:
                for arg in args:
                    try:
                        data = Path(arg).read_bytes()
                        print(hashlib.sha256(data).hexdigest() + "  " + arg)
                    except FileNotFoundError:
                        print(f"sha256sum: {arg}: No such file or directory", file=sys.stderr)
                        sys.exit(1)
            else:
                data = sys.stdin.buffer.read()
                print(hashlib.sha256(data).hexdigest() + "  -")
            """),
        # systemctl: log every invocation; always succeed
        "systemctl": f'#!/bin/sh\necho "$@" >> "{systemctl_log}"\n',
    }
    for name, body in stubs.items():
        p = fake_bin / name
        p.write_text(body)
        p.chmod(0o755)

def test_provision_restarts_containerd_on_first_run(tmp_path):
    """containerd.service is restarted when unit files are written for the first time."""
    fake_bin = tmp_path / "bin"
    fake_bin.mkdir()

    _provision_fake_bin(fake_bin, systemctl_log)

def test_provision_skips_containerd_restart_when_units_unchanged(tmp_path):
    """containerd.service is not restarted when provision writes identical unit files."""
    fake_bin = tmp_path / "bin"
    fake_bin.mkdir()
    systemctl_log = tmp_path / "systemctl.log"
    _provision_fake_bin(fake_bin, systemctl_log)

    # First run — creates units from scratch; restart expected
    r1 = _run_provision(tmp_path, fake_bin)
    assert r1.returncode == 0, r1.stderr

    # Second run — same unit content; containerd restart must be skipped
    systemctl_log.unlink()
    r2 = _run_provision(tmp_path, fake_bin)
    assert r2.returncode == 0, r2.stderr

    calls = systemctl_log.read_text().splitlines() if systemctl_log.exists() else []
    assert not any("restart containerd.service" in c for c in calls), (
        f"containerd was restarted on unchanged provision run:\n" + "\n".join(calls)
    )

#!/usr/bin/env bash
#
# Interactively record the Finder window layout for the release DMG.
#
# Why this script exists
# ----------------------
# When the user double-clicks the released .dmg, Finder opens the volume
# in a window with a custom background image, hidden toolbar/sidebar,
# and Plow.app + Applications icons in fixed positions. None of that
# styling can be set programmatically by hdiutil — Finder reads it from
# a .DS_Store at the volume root. We used to set up that .DS_Store with
# AppleScript at packaging time, but that automation was flaky in CI and
# was removed in favor of shipping a pre-recorded .DS_Store checked in
# at app/plowd/dmg/.DS_Store. plowd-beta-package-signed ditto's that
# file into the staging dir before hdiutil create. So: edits to the
# install-window appearance must be made by re-running this script and
# committing the resulting .DS_Store.
#
# How it works
# ------------
# 1. Stage the same files plowd-beta-package-signed ships: Plow.app,
#    Applications symlink, .background/background.tiff.
# 2. Create a writable (UDRW) DMG and mount it. Finder pops open the
#    volume window; the user arranges it (View Options for background,
#    icon size, arrangement; drag for icon positions and window size).
# 3. User closes the window (Cmd-W), which prompts Finder to flush
#    .DS_Store, and presses Enter at the script's prompt.
# 4. Eject and re-mount. The eject is what guarantees Finder has
#    written .DS_Store to disk; reading it from a still-mounted volume
#    can return a stale or empty file.
# 5. Copy /Volumes/Plow/.DS_Store -> app/plowd/dmg/.DS_Store and
#    detach.
#
# Display-size caveat
# -------------------
# .DS_Store stores the install window's bounds as absolute screen
# coordinates relative to the main display. If you record on a large
# display with the window at (e.g.) 2000,1500, Finder on a small
# laptop will reposition the off-screen window to fit, which loses
# your placement. Record near the top-left of your screen and keep the
# window modest (~680x420 fits every Mac display).
#
# Verifying after recording
# -------------------------
# Before committing, build a release DMG (or stage by hand and run
# hdiutil create -format ULFO with the new .DS_Store ditto'd in) and
# open it to confirm Finder honors the layout. The captured .DS_Store
# should grow noticeably (~6 KB+) when a background picture is set;
# a near-8 KB unchanged size suggests the picture didn't take.
#
# Why a separate script (vs inline in plowd-beta-package-signed)
# --------------------------------------------------------------
# This is a one-off authoring tool with mandatory human-in-Finder
# steps; the release script is unattended CI. Keeping them separate

+#!/usr/bin/env bash
+#
+# Interactively record the Finder window layout for the release DMG.
+#
+# Why this script exists
+# ----------------------
+# When the user double-clicks the released .dmg, Finder opens the volume
+# in a window with a custom background image, hidden toolbar/sidebar,
+# and Plow.app + Applications icons in fixed positions. None of that
+# styling can be set programmatically by hdiutil — Finder reads it from
+# a .DS_Store at the volume root. We used to set up that .DS_Store with
+# AppleScript at packaging time, but that automation was flaky in CI and
+# was removed in favor of shipping a pre-recorded .DS_Store checked in
+# at app/plowd/dmg/.DS_Store. plowd-beta-package-signed ditto's that
+# file into the staging dir before hdiutil create. So: edits to the
+# install-window appearance must be made by re-running this script and
+# committing the resulting .DS_Store.
+#
+# How it works
+# ------------
+# 1. Stage the same files plowd-beta-package-signed ships: Plow.app,
+#    Applications symlink, .background/background.tiff.
+# 2. Create a writable (UDRW) DMG and mount it. Finder pops open the
+#    volume window; the user arranges it (View Options for background,
+#    icon size, arrangement; drag for icon positions and window size).
+# 3. User closes the window (Cmd-W), which prompts Finder to flush
+#    .DS_Store, and presses Enter at the script's prompt.
+# 4. Eject and re-mount. The eject is what guarantees Finder has
+#    written .DS_Store to disk; reading it from a still-mounted volume
+#    can return a stale or empty file.
+# 5. Copy /Volumes/Plow/.DS_Store -> app/plowd/dmg/.DS_Store and
+#    detach.
+#
+# Display-size caveat
+# -------------------
+# .DS_Store stores the install window's bounds as absolute screen
+# coordinates relative to the main display. If you record on a large
+# display with the window at (e.g.) 2000,1500, Finder on a small
+# laptop will reposition the off-screen window to fit, which loses
+# your placement. Record near the top-left of your screen and keep the
+# window modest (~680x420 fits every Mac display).
+#
+# Verifying after recording
+# -------------------------
+# Before committing, build a release DMG (or stage by hand and run
+# hdiutil create -format ULFO with the new .DS_Store ditto'd in) and
+# open it to confirm Finder honors the layout. The captured .DS_Store
+# should grow noticeably (~6 KB+) when a background picture is set;
+# a near-8 KB unchanged size suggests the picture didn't take.
+#
+# Why a separate script (vs inline in plowd-beta-package-signed)
+# --------------------------------------------------------------
+# This is a one-off authoring tool with mandatory human-in-Finder
+# steps; the release script is unattended CI. Keeping them separate
+# makes the release path deterministic (just `ditto` a checked-in
+# blob) and lets layout changes go through a normal code review of
+# the .DS_Store diff.
+set -euo pipefail
+
+usage() {
+  cat <<'EOF'
+Usage: plowd-record-dmg-layout [--app <path>]
+
+Records app/plowd/dmg/.DS_Store from an interactive Finder layout
+session. Mounts a writable scratch DMG that mirrors the staging
+plowd-beta-package-signed ships, drops you into Finder to arrange the
+install window, then captures the resulting .DS_Store back into the
+repo when you press Enter at the prompt.
+
+Workflow once mounted at /Volumes/Plow:
+  1. View > as Icons (Cmd-1)
+  2. View > Hide Toolbar (Opt-Cmd-T)
+  3. View > Show View Options (Cmd-J)
+       - size/position the window (keep it under ~680x420 and near
+         the top-left of your screen so it fits on small displays)
+       - drag .background/background.tiff into Background > Picture
+         (open a second Finder window, Cmd-Shift-. to show hidden
+         files, then drag the .tiff into the well; the file picker
+         is unreliable for hidden paths)
+       - icon size, text size, arrangement = None
+  4. Drag Plow.app and Applications icons into position
+  5. Close the volume window (Cmd-W) to flush .DS_Store
+  6. Press Enter at this script's prompt to capture
+
+Options:
+  --app <path>   Path to the .app to stage. Defaults to the release
+                 build at .build/xcode-derived/Build/Products/Release/
+                 Plow.app, falling back to /Applications/Plow.app
+                 (which is fine for layout-only iteration since icon
+                 positions are recorded by filename).
+  -h, --help     Show this message.
+EOF
+}
+
+repo_root="$(cd "$(dirname "${BASH_SOURCE[0]}")/../../.." && pwd)"
+app_path=""
+
+while [[ $# -gt 0 ]]; do
+  case "$1" in
+    --app)
+      app_path="${2:-}"
+      shift 2
+      ;;
+    -h|--help)
+      usage
+      exit 0
+      ;;
+    *)
+      echo "error: unknown argument: $1" >&2
+      usage >&2
+      exit 1
+      ;;
+  esac
+done
+
+if [[ -z "${app_path}" ]]; then
+  release_app="${repo_root}/.build/xcode-derived/Build/Products/Release/Plow.app"
+  if [[ -d "${release_app}" ]]; then
+    app_path="${release_app}"
+  elif [[ -d "/Applications/Plow.app" ]]; then
+    app_path="/Applications/Plow.app"
+    echo "note: using /Applications/Plow.app (no release build found)"
+  else
+    echo "error: no Plow.app found; pass --app <path>, run 'just app release', or install via /plow-prod-install" >&2
+    exit 1
+  fi
+fi
+[[ -d "${app_path}" ]] || { echo "error: Plow.app not found at ${app_path}" >&2; exit 1; }
+
+bg_src="${repo_root}/app/plowd/dmg/background.tiff"
+ds_dest="${repo_root}/app/plowd/dmg/.DS_Store"
+[[ -f "${bg_src}" ]] || { echo "error: background not found: ${bg_src}" >&2; exit 1; }
+
+mounted_plow_volumes="$(hdiutil info 2>/dev/null | grep -Eo '/Volumes/Plow( [0-9]+)?$' || true)"
+mounted_plow_volumes="$(printf '%s\n' "${mounted_plow_volumes}" | sort -u)"
+if [[ -n "${mounted_plow_volumes}" ]]; then
+  echo "error: a Plow volume is already mounted; eject it before recording:" >&2
+  printf '%s\n' "${mounted_plow_volumes}" >&2
+  exit 1
+fi
+
+work_dir="$(mktemp -d /tmp/plowd-dmg-record.XXXXXX)"
+stage_dir="${work_dir}/stage"
+rw_dmg="${work_dir}/layout.dmg"
+mounted_volume=""
+cleanup() {
+  if [[ -n "${mounted_volume}" ]]; then
+    hdiutil detach "${mounted_volume}" >/dev/null 2>&1 || true
+  fi
+  rm -rf "${work_dir}"
+}
+trap cleanup EXIT
+
+mkdir -p "${stage_dir}/.background"
+ditto "${app_path}" "${stage_dir}/Plow.app"
+ditto "${bg_src}" "${stage_dir}/.background/background.tiff"
+ln -s /Applications "${stage_dir}/Applications"
+
+hdiutil create -volname "Plow" -srcfolder "${stage_dir}" -ov -format UDRW "${rw_dmg}" >/dev/null
+attach_output="$(hdiutil attach "${rw_dmg}" -readwrite -noverify)"
+mounted_volume="$(printf '%s\n' "${attach_output}" | sed -n 's|.*\(/Volumes/.*\)$|\1|p' | head -n 1)"
+[[ -n "${mounted_volume}" ]] || { echo "error: failed to determine mounted volume for ${rw_dmg}" >&2; exit 1; }
+
+cat <<EOF
+
+mounted at ${mounted_volume}
+
+In Finder:
+  - View > as Icons (Cmd-1)
+  - View > Hide Toolbar (Opt-Cmd-T)
+  - View > Show View Options (Cmd-J)
+      * size/position the window
+      * drag .background/background.tiff into Background > Picture
+        (Cmd-Shift-. in the picker shows hidden files)
+      * set icon size, text size, arrangement = None
+  - position Plow.app and Applications icons
+  - close the window (Cmd-W) to flush .DS_Store
+
+EOF
+printf "Press Enter when done... "
+read -r _ </dev/tty
+
+hdiutil detach "${mounted_volume}" >/dev/null
+mounted_volume=""
+
+attach_output="$(hdiutil attach "${rw_dmg}" -readwrite -noverify -noautoopen)"
+mounted_volume="$(printf '%s\n' "${attach_output}" | sed -n 's|.*\(/Volumes/.*\)$|\1|p' | head -n 1)"
+[[ -n "${mounted_volume}" ]] || { echo "error: failed to remount ${rw_dmg}" >&2; exit 1; }
+
+[[ -f "${mounted_volume}/.DS_Store" ]] || {
+  echo "error: no .DS_Store in mounted volume — did Finder finalize the layout (Cmd-W)?" >&2
+  exit 1
+}
+
+cp "${mounted_volume}/.DS_Store" "${ds_dest}"
+hdiutil detach "${mounted_volume}" >/dev/null
+mounted_volume=""
+
+echo
+echo "captured ${ds_dest}"
+git -C "${repo_root}" diff --stat -- "${ds_dest}" 2>/dev/null || true

def test_validate_package_rejects_bundle_missing_plowd_dist_info(tmp_path):
    """A packaged runtime that exists structurally but lacks plowd-*.dist-info/
    is the exact regression #478 shape: site-packages/ is present, all files
    pass the existence check, but plowd dies at import with PackageNotFoundError
    because importlib.metadata can't find the metadata."""
    app_bundle = tmp_path / "Plow.app"
    _create_minimal_valid_app_bundle(app_bundle)

    # Sanity: the unmodified bundle passes validation.
    ok = subprocess.run(
        [str(VALIDATE_PACKAGE_SCRIPT), "--app-bundle", str(app_bundle), "--mode", "offline"],
        capture_output=True,
        text=True,
    )
    assert ok.returncode == 0, f"baseline bundle should validate cleanly, got: {ok.stderr}"

    # Remove the dist-info; site-packages/ itself stays so this isn't caught
    # by the existing required_runtime_paths loop.
    dist_info = (
        app_bundle / "Contents" / "Resources" / "runtime"
        / "site-packages" / "plowd-0.1.1.dist-info"
    )
    import shutil
    shutil.rmtree(dist_info)

    fail = subprocess.run(
        [str(VALIDATE_PACKAGE_SCRIPT), "--app-bundle", str(app_bundle), "--mode", "offline"],
        capture_output=True,
        text=True,
    )
    assert fail.returncode != 0, "validate-package must reject a bundle missing plowd dist-info"
    assert "plowd-*.dist-info" in fail.stderr, (
        f"error message should mention the missing dist-info; got stderr:\n{fail.stderr}"
    )
