+#!/usr/bin/env bash
+# Mount a Plow DMG and assert it has no custom volume icon (PLO-29 regression
+# guard) and that the volume name is "Plow". Detaches before exiting.
+set -euo pipefail
+
+usage() {
+  cat <<'EOF'
+Usage: plowd-verify-dmg [--dmg <path>]
+
+Mounts the given DMG, asserts the mounted volume is named "Plow" and has no
+.VolumeIcon.icns at its root, then detaches. Defaults to .build/Plow.dmg
+relative to the repo root.
+EOF
+}
+
+repo_root="$(cd "$(dirname "${BASH_SOURCE[0]}")/../../.." && pwd)"
+dmg_path="${repo_root}/.build/Plow.dmg"
+
+while [[ $# -gt 0 ]]; do
+  case "$1" in
+    --dmg)
+      dmg_path="${2:-}"
+      shift 2
+      ;;
+    -h|--help)
+      usage
+      exit 0
+      ;;
+    *)
+      echo "error: unknown argument: $1" >&2
+      usage >&2
+      exit 1
+      ;;
+  esac
+done
+
+if [[ "${dmg_path}" != /* ]]; then
+  dmg_path="${repo_root}/${dmg_path}"
+fi
+
+[[ -f "${dmg_path}" ]] || { echo "error: DMG not found: ${dmg_path}" >&2; exit 1; }
+
+# Refuse to run when a Plow volume is already mounted: the new mountpoint
+# would be "Plow 1" / "Plow 2" / ..., which would mask a real regression in
+# the volume-name assertion below.
+mounted_plow_volumes="$(hdiutil info 2>/dev/null | grep -Eo '/Volumes/Plow( [0-9]+)?$' || true)"
+mounted_plow_volumes="$(printf '%s\n' "${mounted_plow_volumes}" | sort -u)"
+if [[ -n "${mounted_plow_volumes}" ]]; then
+  echo "error: a Plow volume is already mounted; eject it before verifying:" >&2
+  printf '%s\n' "${mounted_plow_volumes}" >&2
+  exit 1
+fi
+
+mounted_volume=""
+cleanup() {
+  local rc=$?
+  if [[ -n "${mounted_volume}" ]]; then
+    if [[ "${rc}" -eq 0 ]]; then
+      # Success path: detach failure (e.g. volume busy) must be fatal.
+      # Otherwise the next release retry trips ensure_no_mounted_plow_volume
+      # in plowd-beta-package-signed and aborts on a leaked mount.
+      if ! hdiutil detach "${mounted_volume}"; then
+        echo "error: failed to detach ${mounted_volume} after successful verify" >&2
+        exit 1
+      fi
+    else
+      # Failure path: don't let detach noise mask the real error.
+      hdiutil detach "${mounted_volume}" >/dev/null 2>&1 || true
+    fi
+  fi
+  exit "${rc}"
+}
+trap cleanup EXIT
+
+attach_output="$(hdiutil attach "${dmg_path}" -nobrowse -readonly)"
+mounted_volume="$(printf '%s\n' "${attach_output}" | sed -n 's|.*\(/Volumes/.*\)$|\1|p' | head -n 1)"

+

+#!/usr/bin/env bash
+#
+# Interactively record the Finder window layout for the release DMG.
+#
+# Why this script exists
+# ----------------------
+# When the user double-clicks the released .dmg, Finder opens the volume
+# in a window with a custom background image, hidden toolbar/sidebar,
+# and Plow.app + Applications icons in fixed positions. None of that
+# styling can be set programmatically by hdiutil — Finder reads it from
+# a .DS_Store at the volume root. We used to set up that .DS_Store with
+# AppleScript at packaging time, but that automation was flaky in CI and
+# was removed in favor of shipping a pre-recorded .DS_Store checked in
+# at app/plowd/dmg/.DS_Store. plowd-beta-package-signed ditto's that
+# file into the staging dir before hdiutil create. So: edits to the
+# install-window appearance must be made by re-running this script and
+# committing the resulting .DS_Store.
+#
+# How it works
+# ------------
+# 1. Stage the same files plowd-beta-package-signed ships: Plow.app,
+#    Applications symlink, .background/background.tiff.
+# 2. Create a writable (UDRW) DMG and mount it. Finder pops open the
+#    volume window; the user arranges it (View Options for background,
+#    icon size, arrangement; drag for icon positions and window size).
+# 3. User closes the window (Cmd-W), which prompts Finder to flush
+#    .DS_Store, and presses Enter at the script's prompt.
+# 4. Eject and re-mount. The eject is what guarantees Finder has
+#    written .DS_Store to disk; reading it from a still-mounted volume
+#    can return a stale or empty file.
+# 5. Copy /Volumes/Plow/.DS_Store -> app/plowd/dmg/.DS_Store and
+#    detach.
+#
+# Display-size caveat
+# -------------------
+# .DS_Store stores the install window's bounds as absolute screen
+# coordinates relative to the main display. If you record on a large
+# display with the window at (e.g.) 2000,1500, Finder on a small
+# laptop will reposition the off-screen window to fit, which loses
+# your placement. Record near the top-left of your screen and keep the
+# window modest (~680x420 fits every Mac display).
+#
+# Verifying after recording
+# -------------------------
+# Before committing, build a release DMG (or stage by hand and run
+# hdiutil create -format ULFO with the new .DS_Store ditto'd in) and
+# open it to confirm Finder honors the layout. The captured .DS_Store
+# should grow noticeably (~6 KB+) when a background picture is set;
+# a near-8 KB unchanged size suggests the picture didn't take.
+#
+# Why a separate script (vs inline in plowd-beta-package-signed)
+# --------------------------------------------------------------
+# This is a one-off authoring tool with mandatory human-in-Finder
+# steps; the release script is unattended CI. Keeping them separate
+# makes the release path deterministic (just `ditto` a checked-in
+# blob) and lets layout changes go through a normal code review of
+# the .DS_Store diff.
+set -euo pipefail
+
+usage() {
+  cat <<'EOF'
+Usage: plowd-record-dmg-layout [--app <path>]
+
+Records app/plowd/dmg/.DS_Store from an interactive Finder layout
+session. Mounts a writable scratch DMG that mirrors the staging
+plowd-beta-package-signed ships, drops you into Finder to arrange the
+install window, then captures the resulting .DS_Store back into the
+repo when you press Enter at the prompt.
+
