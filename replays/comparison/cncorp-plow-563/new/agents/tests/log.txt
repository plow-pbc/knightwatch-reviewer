     5	/// Holds `kIOPMAssertionTypePreventSystemSleep` while the user has opted in, so
     6	/// the system stays in Dark Wake and the OpenClaw VM keeps serving through
     7	/// display sleep. On battery the kernel effectively downgrades the assertion to
     8	/// idle-sleep-prevention, so a single assertion works in both power states;
     9	/// draining the battery is the user's explicit choice when they toggle this on.
    10	/// Clamshell sleep on a MacBook without an external display is still enforced
    11	/// by the kernel and isn't reachable from a public API.
    12	@Observable
    13	@MainActor
    14	final class KeepMacAwake {
    15	    static let preferencesKey = "keepMacAwakeWhileRunning"
    16	
    17	    private var assertionID: IOPMAssertionID = 0
    18	    private var hasAssertion = false
    19	    /// Set true while we're rewriting `isEnabled` from inside `applyEnabled` to
    20	    /// revert a failed acquire — the recursive `didSet` reads this and bails.
    21	    private var suppressApply = false
    22	
    23	    var isEnabled: Bool = UserDefaults.standard.bool(forKey: KeepMacAwake.preferencesKey) {
    24	        didSet { applyEnabled() }
    25	    }
    26

    29	    }
    30	
    31	    /// Releases the assertion deterministically before the app exits, so the
    32	    /// system can coalesce any pending sleep state at logout/shutdown without
    33	    /// waiting for process reaping. The kernel reclaims the assertion on
    34	    /// process exit either way; this is a courtesy, not a leak fix.
    35	    func teardown() {
    36	        releaseAssertion()
    37	    }
    38	
    39	    /// Brings the held IOPMAssertion in line with `isEnabled` and persists the
    40	    /// resolved value. If acquire fails, `isEnabled` is reverted to false so
    41	    /// the toggle UI matches the actual sleep-prevention state.
    42	    private func applyEnabled() {

    44	        if isEnabled {
    45	            if !acquireAssertion() {
    46	                suppressApply = true
    47	                isEnabled = false
    48	                suppressApply = false
    49	            }
    50	        } else {
    51	            releaseAssertion()
    52	        }
    53	        UserDefaults.standard.set(isEnabled, forKey: Self.preferencesKey)
    54	    }
