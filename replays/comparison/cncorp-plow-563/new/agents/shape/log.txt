+@testable import Phoenix

+@MainActor
+final class KeepMacAwakeTests: XCTestCase {
+    private var defaults: CountingUserDefaults!
+    private var suiteName: String!
+
+    override func setUp() {
+        super.setUp()
+        suiteName = "KeepMacAwakeTests-\(UUID().uuidString)"
+        defaults = CountingUserDefaults(suiteName: suiteName)!
+        defaults.removePersistentDomain(forName: suiteName)
+    }
+
+    override func tearDown() {
+        defaults.removePersistentDomain(forName: suiteName)
+        defaults = nil
+        suiteName = nil
+        super.tearDown()
+    }
+
+    // Toggle ON → IOPMAssertionCreateWithName fails → isEnabled must revert to
+    // false, UserDefaults must persist false, acquire must be tried exactly
+    // once (no recursion through suppressApply), and exactly one UserDefaults
+    // write of the final value must occur for the toggle interaction.
+    func testFailedAcquireRevertsToggleAndWritesOnceWithoutRecursion() {
+        defaults.set(false, forKey: KeepMacAwake.preferencesKey)
+
+        var acquireCallCount = 0
+        let failingAcquire: KeepMacAwake.AssertionCreator = {
+            acquireCallCount += 1
+            return (kIOReturnError, IOPMAssertionID(0))
+        }
+
+        let keepMacAwake = KeepMacAwake(userDefaults: defaults, assertionCreator: failingAcquire)
+        // After init: isEnabled=false, applyEnabled goes to else branch
+        // (releaseAssertion no-op) and writes false to UserDefaults. Reset
+        // the test counters AFTER init to scope the assertions to the
+        // toggle interaction.
+        XCTAssertFalse(keepMacAwake.isEnabled)
+        XCTAssertEqual(acquireCallCount, 0, "init should not attempt acquire when seeded false")
+        defaults.resetWriteCount(forKey: KeepMacAwake.preferencesKey)
+
+        // The pre-fix code would have left isEnabled=true and UserDefaults=true
+        // after this assignment because the failure path silently swallowed
+        // the IOReturn. The post-fix code reverts both, in a single bounded
+        // didSet round.
+        keepMacAwake.isEnabled = true
+
+        XCTAssertFalse(keepMacAwake.isEnabled, "isEnabled must revert to false when acquire fails")
+        XCTAssertFalse(
+            defaults.bool(forKey: KeepMacAwake.preferencesKey),
+            "UserDefaults must persist the reverted false, not the user's intended true"
+        )
+        XCTAssertEqual(
+            acquireCallCount, 1,
+            "acquireAssertion must be called exactly once — suppressApply gate must prevent the inner isEnabled=false write from re-entering applyEnabled and re-invoking the closure"
+        )
+        XCTAssertEqual(
