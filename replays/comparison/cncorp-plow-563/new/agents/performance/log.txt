import Foundation
import IOKit.pwr_mgt
import Observation

/// Holds `kIOPMAssertionTypePreventSystemSleep` while the user has opted in, so
/// the system stays in Dark Wake and the OpenClaw VM keeps serving through
/// display sleep. On battery the kernel effectively downgrades the assertion to
/// idle-sleep-prevention, so a single assertion works in both power states;
/// draining the battery is the user's explicit choice when they toggle this on.
/// Clamshell sleep on a MacBook without an external display is still enforced
/// by the kernel and isn't reachable from a public API.
@Observable
@MainActor
final class KeepMacAwake {
    static let preferencesKey = "keepMacAwakeWhileRunning"

    private var assertionID: IOPMAssertionID = 0
    private var hasAssertion = false
    /// Set true while we're rewriting `isEnabled` from inside `applyEnabled` to
    /// revert a failed acquire — the recursive `didSet` reads this and bails.
    private var suppressApply = false

    var isEnabled: Bool = UserDefaults.standard.bool(forKey: KeepMacAwake.preferencesKey) {
        didSet { applyEnabled() }
    }

    init() {
        applyEnabled()
    }

    /// Releases the assertion deterministically before the app exits, so the
    /// system can coalesce any pending sleep state at logout/shutdown without
    /// waiting for process reaping. The kernel reclaims the assertion on
    /// process exit either way; this is a courtesy, not a leak fix.
    func teardown() {
        releaseAssertion()
    }

    /// Brings the held IOPMAssertion in line with `isEnabled` and persists the
    /// resolved value. If acquire fails, `isEnabled` is reverted to false so
    /// the toggle UI matches the actual sleep-prevention state.
    private func applyEnabled() {
        guard !suppressApply else { return }
        if isEnabled {
            if !acquireAssertion() {
                suppressApply = true
                isEnabled = false
                suppressApply = false
            }
        } else {
            releaseAssertion()
        }
        UserDefaults.standard.set(isEnabled, forKey: Self.preferencesKey)
    }

    private func acquireAssertion() -> Bool {
        guard !hasAssertion else { return true }
        var id: IOPMAssertionID = 0
        let reason = "Plow keeping Mac awake so the OpenClaw VM stays reachable" as CFString
        let result = IOPMAssertionCreateWithName(
            kIOPMAssertionTypePreventSystemSleep as CFString,
            IOPMAssertionLevel(kIOPMAssertionLevelOn),
            reason,
            &id
        )
        guard result == kIOReturnSuccess else {
            NSLog("[KeepMacAwake] IOPMAssertionCreateWithName failed: %d", result)
            return false
        }
        assertionID = id
        hasAssertion = true
        return true
    }

    private func releaseAssertion() {
        guard hasAssertion else { return }
        IOPMAssertionRelease(assertionID)
        hasAssertion = false
        assertionID = 0
    }
}

    private func updateWindowHeight(for screen: InstallerScreen) {
        guard let window = NSApp.windows.first(where: { $0.identifier?.rawValue == "installer" }) else { return }
        let targetHeight = targetWindowHeight(for: screen)
        let currentFrame = window.frame
        guard abs(currentFrame.height - targetHeight) > 1 else { return }
        let newOriginY = currentFrame.origin.y + (currentFrame.height - targetHeight)
        let newFrame = NSRect(
            x: currentFrame.origin.x,
            y: newOriginY,
            width: currentFrame.width,
            height: targetHeight
        )
        NSAnimationContext.runAnimationGroup { context in
            context.duration = 0.45
            context.timingFunction = CAMediaTimingFunction(name: .easeOut)
            window.animator().setFrame(newFrame, display: true)
        }
    }

    /// The connectors screen sizes its window to the measured content height
    /// (reported via `InstallerContentHeightKey`), capped by the visible
    /// screen so a long settings list falls back to scrolling inside the
    /// existing `ScrollView` instead of pushing the window off-screen. The
    /// remaining screens are intentionally taller than their content
    /// (Spacer-filled splash layouts) and keep the design height of 530.
    private func targetWindowHeight(for screen: InstallerScreen) -> CGFloat {
        let cap = max(420, (NSScreen.main?.visibleFrame.height ?? 900) - 80)
        if screen == .connectors, measuredContentHeight > 0 {
            // `measuredContentHeight` is the ScrollView's inner content; add
            // the window's titlebar plus the bottom-anchored DownloadBarView's
            // measured height (zero when the bar isn't shown) so the bar
            // doesn't cover the last settings rows.
            return min(measuredContentHeight + 28 + measuredOverlayHeight, cap)
        }
        return 530
    }

/// Carries the bottom-anchored `DownloadBarView`'s measured height up to
/// `InstallerView` so the connectors-screen window can grow to keep the

/// Reverts to zero when the bar isn't in the view tree (post-install).
struct InstallerOverlayHeightKey: PreferenceKey {
    static var defaultValue: CGFloat = 0
    static func reduce(value: inout CGFloat, nextValue: () -> CGFloat) {
        value = nextValue()

/// One-time cleanup of artifacts written by the pre-Lima (VZ) runtime. Existing
/// users upgrading from a VZ build carry 200MB–1GB of dead disk Lima never
/// touches. These paths are fully owned by the removed VMManager /
/// GatekeeperVMManager and have no live readers under the Lima runtime, so
/// unconditional removal is safe. Idempotent: absent paths are a no-op.
///
/// Targeted paths (all under `<appSupport>/`):
///   - `vm/rootfs.img` — writable copy of the OpenClaw VZ disk image.
///   - `gatekeeper-vm-state/` — VZ Gatekeeper instance directory (rootfs +
///     metadata), named by the removed GatekeeperVMManager.
///   - `gatekeeper-console.log` — VZ Gatekeeper serial-console log.
///
/// Downloaded gatekeeper tarballs in `runtime-cache/` are already reaped by
/// `RuntimeCacheCleaner.prune`, which drops any file not named in the current
/// manifest — so they don't need to be listed here.
enum LegacyVZArtifactCleanup {
    static func run() {
        run(appSupportRoot: AppPaths.appSupportRoot())
    }

    static func run(appSupportRoot: URL) {
        let fm = FileManager.default
        let targets: [URL] = [
            appSupportRoot.appendingPathComponent("vm/rootfs.img", isDirectory: false),
            appSupportRoot.appendingPathComponent("gatekeeper-vm-state", isDirectory: true),
            appSupportRoot.appendingPathComponent("gatekeeper-console.log", isDirectory: false),
        ]
        for url in targets {
            guard fm.fileExists(atPath: url.path) else { continue }
            let size = (try? fm.attributesOfItem(atPath: url.path)[.size] as? Int64) ?? -1
            do {
                try fm.removeItem(at: url)
                NSLog("[LegacyCleanup] removed stale VZ artifact path=%@ size=%lld", url.path, size)
            } catch {
                NSLog("[LegacyCleanup] failed to remove path=%@ error=%@", url.path, error.localizedDescription)
            }
        }
    }
}

+        defaults.set(false, forKey: KeepMacAwake.preferencesKey)

+            acquireCallCount += 1

+        // toggle interaction.

    46	        applyEnabled()
    47	    }

    49	    /// Releases the assertion deterministically before the app exits, so the
    50	    /// system can coalesce any pending sleep state at logout/shutdown without
    51	    /// waiting for process reaping. The kernel reclaims the assertion on
    52	    /// process exit either way; this is a courtesy, not a leak fix.
    53	    func teardown() {
    54	        releaseAssertion()
    55	    }
    56	
    57	    /// Brings the held IOPMAssertion in line with `isEnabled` and persists the
    58	    /// resolved value. If acquire fails, `isEnabled` is reverted to false so
    59	    /// the toggle UI matches the actual sleep-prevention state.
    60	    private func applyEnabled() {

    62	        if isEnabled {
    63	            if !acquireAssertion() {
    64	                suppressApply = true
    65	                isEnabled = false
    66	                suppressApply = false
    67	            }
    68	        } else {
    69	            releaseAssertion()
    70	        }
    71	        userDefaults.set(isEnabled, forKey: Self.preferencesKey)
    72	    }

    75	        guard !hasAssertion else { return true }
    76	        let outcome = assertionCreator()

    79	            return false
    80	        }
    81	        assertionID = outcome.assertionID
    82	        hasAssertion = true
    83	        return true
    84	    }
    85	
    86	    private func releaseAssertion() {
    87	        guard hasAssertion else { return }
    88	        IOPMAssertionRelease(assertionID)
    89	        hasAssertion = false
    90	        assertionID = 0
    91	    }
