+    // Toggle ON → IOPMAssertionCreateWithName fails → isEnabled must revert to
+    // false, UserDefaults must persist false, acquire must be tried exactly

+    // write of the final value must occur for the toggle interaction.
+    func testFailedAcquireRevertsToggleAndWritesOnceWithoutRecursion() {
+        defaults.set(false, forKey: KeepMacAwake.preferencesKey)
+
+        var acquireCallCount = 0
+        let failingAcquire: KeepMacAwake.AssertionCreator = {
+            acquireCallCount += 1
+            return (kIOReturnError, IOPMAssertionID(0))

+        // After init: isEnabled=false, applyEnabled goes to else branch
+        // (releaseAssertion no-op) and writes false to UserDefaults. Reset
+        // the test counters AFTER init to scope the assertions to the
+        // toggle interaction.
+        XCTAssertFalse(keepMacAwake.isEnabled)
