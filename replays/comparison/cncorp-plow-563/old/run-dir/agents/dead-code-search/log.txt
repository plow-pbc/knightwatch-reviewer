import Foundation
import IOKit.pwr_mgt
import Observation

/// Holds `kIOPMAssertionTypePreventSystemSleep` while the user has opted in, so
/// the system stays in Dark Wake and the OpenClaw VM keeps serving through
/// display sleep. On battery the kernel effectively downgrades the assertion to
/// idle-sleep-prevention, so a single assertion works in both power states;
/// draining the battery is the user's explicit choice when they toggle this on.
/// Clamshell sleep on a MacBook without an external display is still enforced
/// by the kernel and isn't reachable from a public API.
@Observable
@MainActor
final class KeepMacAwake {
    static let preferencesKey = "keepMacAwakeWhileRunning"

    private var assertionID: IOPMAssertionID = 0
    private var hasAssertion = false
    /// Set true while we're rewriting `isEnabled` from inside `applyEnabled` to
    /// revert a failed acquire — the recursive `didSet` reads this and bails.
    private var suppressApply = false

    var isEnabled: Bool = UserDefaults.standard.bool(forKey: KeepMacAwake.preferencesKey) {
        didSet { applyEnabled() }
    }

    init() {
        applyEnabled()
    }

    /// Releases the assertion deterministically before the app exits, so the
    /// system can coalesce any pending sleep state at logout/shutdown without
    /// waiting for process reaping. The kernel reclaims the assertion on
    /// process exit either way; this is a courtesy, not a leak fix.
    func teardown() {
        releaseAssertion()
    }

    /// Brings the held IOPMAssertion in line with `isEnabled` and persists the
    /// resolved value. If acquire fails, `isEnabled` is reverted to false so
    /// the toggle UI matches the actual sleep-prevention state.
    private func applyEnabled() {
        guard !suppressApply else { return }
        if isEnabled {
            if !acquireAssertion() {
                suppressApply = true
                isEnabled = false
                suppressApply = false
            }
        } else {
            releaseAssertion()
        }
        UserDefaults.standard.set(isEnabled, forKey: Self.preferencesKey)
    }

    private func acquireAssertion() -> Bool {
        guard !hasAssertion else { return true }
        var id: IOPMAssertionID = 0
        let reason = "Plow keeping Mac awake so the OpenClaw VM stays reachable" as CFString
        let result = IOPMAssertionCreateWithName(
            kIOPMAssertionTypePreventSystemSleep as CFString,
            IOPMAssertionLevel(kIOPMAssertionLevelOn),
            reason,
            &id
        )
        guard result == kIOReturnSuccess else {
            NSLog("[KeepMacAwake] IOPMAssertionCreateWithName failed: %d", result)
            return false
        }
        assertionID = id
        hasAssertion = true
        return true
    }

    private func releaseAssertion() {
        guard hasAssertion else { return }
        IOPMAssertionRelease(assertionID)
        hasAssertion = false
        assertionID = 0
    }
}

/// Owns the app's menu-bar presence as a real NSStatusItem + NSPanel.
///
/// We used MenuBarExtra originally, but macOS silently hides the MenuBarExtra

///
