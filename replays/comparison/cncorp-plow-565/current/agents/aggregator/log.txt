   391	/// to the SwiftUI content's intrinsic height. We drive the window frame
   392	/// directly rather than going through `setContentSize`, because the latter
   393	/// treats the SwiftUI ideal as *content-area* size and then adds the titlebar

   397	    /// When true, intrinsic-driven resizes (after the initial sizing) animate
   398	    /// via AppKit's setFrame animator. The menu-bar panel keeps this off
   399	    /// because it has its own fade orchestration; the installer window flips

   403	    /// When true, the post-resize frame is clamped to the host screen's
   404	    /// `visibleFrame` so growth (which pins the top edge and pushes the
   405	    /// bottom down) cannot slide the window under the Dock or above the
   406	    /// menu bar. The menu-bar panel keeps this off because it's anchored
   407	    /// to the menu bar by construction; floating windows that the user
   408	    /// repositions (the installer is `window.center()`-ed at open) need
   409	    /// the clamp because top-edge pinning otherwise compounds with growth
   410	    /// to march the window off-screen on tall content / small displays.

   416	        super.invalidateIntrinsicContentSize()
   417	        syncWindowSize(animate: animatesWindowResize)
   418	    }

   421	        super.viewDidMoveToWindow()
   422	        // Initial sizing (window just opened) should never animate — there's
   423	        // no prior frame to animate from.

   427	    private func syncWindowSize(animate: Bool) {
   428	        guard let window else { return }
   429	        let size = intrinsicContentSize
   430	        guard size.width > 0, size.height > 0 else { return }
   431	        if window.frame.size == size { return }
   432	        // Pin the top edge — both the menu-bar panel and the installer window
   433	        // anchor visually at the top, so growth must expand downward.
   434	        var frame = window.frame
   435	        frame.origin.y += frame.size.height - size.height
   436	        frame.size = size
   437	        if clampsToScreen, let visibleFrame = (window.screen ?? NSScreen.main)?.visibleFrame {
   438	            // Fix the top edge first, then the bottom — order matters so
   439	            // the bottom adjustment wins when the window is taller than
   440	            // the visible frame (degenerate; cap is supposed to prevent
   441	            // it). Result in that case: top pinned to visibleFrame.minY,
   442	            // bottom may extend above visibleFrame.maxY.
   443	            if frame.maxY > visibleFrame.maxY {
   444	                frame.origin.y = visibleFrame.maxY - frame.height
   445	            }
   446	            if frame.minY < visibleFrame.minY {
   447	                frame.origin.y = visibleFrame.minY
   448	            }
   449	        }
   450	        window.setFrame(frame, display: true, animate: animate)
   451	    }

+/// the existing ScrollView fallback re-engages on overflow.
+final class InstallerWindowSizeTests: XCTestCase {

+    private let designHeight: CGFloat = 530
+    private let cap: CGFloat = 800
+
+    func testSplashScreensUseDesignHeight() {
+        for screen in [InstallerScreen.welcomeInstall, .welcomeFDA, .activation, .connectContext] {
+            XCTAssertEqual(
+                installerWindowHeight(for: screen, contentHeight: 1200, overlayHeight: 200, cap: cap),
+                designHeight,
+                "splash screen \(screen) must ignore measured content and stay at design height"
+            )
+        }
+    }
+
+    func testConnectorsWithoutMeasurementFallsBackToDesignHeight() {
+        XCTAssertEqual(
+            installerWindowHeight(for: .connectors, contentHeight: 0, overlayHeight: 0, cap: cap),
+            designHeight,
+            "before content measures, render at the splash height — never zero"
+        )
+    }
+
+    func testConnectorsSizesToMeasuredContentBelowCap() {
+        XCTAssertEqual(
+            installerWindowHeight(for: .connectors, contentHeight: 480, overlayHeight: 0, cap: cap),
+            480
+        )
+    }
+
+    func testConnectorsAddsOverlayHeightToContent() {
+        // The DownloadBarView overlays the bottom of the ZStack — add its
+        // measured height so settings rows aren't hidden behind it.
+        XCTAssertEqual(

+            550
+        )
+    }
+
+    func testConnectorsCapsAtVisibleScreen() {
+        // Content + overlay would push past the cap → cap wins so the
+        // existing ScrollView fallback inside SettingsView re-engages.
+        XCTAssertEqual(
+            installerWindowHeight(for: .connectors, contentHeight: 1500, overlayHeight: 100, cap: cap),
+            cap

+    }

+    func testConnectorsRetryStateGrowsWindowWithBar() {

+        // Assert exact values so a sign-flip in the helper still trips.

+            400
+        )
