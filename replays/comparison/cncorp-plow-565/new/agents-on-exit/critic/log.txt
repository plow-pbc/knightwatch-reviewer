/// activation / connectContext) keep the design height of 530 because they
/// fill via `Spacer`s rather than intrinsic content. The connectors screen
/// sizes to its inner SettingsView content (reported via
/// `InstallerContentHeightKey`) plus the bottom-anchored DownloadBarView's
/// height (reported via `InstallerOverlayHeightKey`, zero when absent), and
/// is capped so a long settings list falls back to scrolling inside
/// SettingsView's existing ScrollView rather than pushing the window
/// off-screen.
///
/// Returns 530 (the splash height) for connectors when no measurement has
/// arrived yet — better to render at the design height for one frame than
/// flash a degenerate window.
///
/// The cap applies to *all* screens, including splash: on a degenerate
/// `visibleFrame` (multi-display reconfiguration, sidecar with a tiny
/// secondary display, etc.) the cap may drop below 530, in which case
/// fitting the visible frame matters more than honoring the design height.
func installerWindowHeight(
    for screen: InstallerScreen,
    contentHeight: CGFloat,
    overlayHeight: CGFloat,
    cap: CGFloat
) -> CGFloat {
    let splashHeight: CGFloat = 530
    let unclamped: CGFloat = {
        guard screen == .connectors, contentHeight > 0 else { return splashHeight }
        return contentHeight + overlayHeight
    }()
    return min(unclamped, cap)
}

/// doesn't inflate SwiftUI's ideal size) and keeps its hosting window sized
/// to the SwiftUI content's intrinsic height. We drive the window frame
/// directly rather than going through `setContentSize`, because the latter
/// treats the SwiftUI ideal as *content-area* size and then adds the titlebar
/// back on — producing a window that is taller than the content by exactly
/// the titlebar height.
final class SelfSizingHostingView<Content: View>: NSHostingView<Content> {
    /// When true, intrinsic-driven resizes (after the initial sizing) animate
    /// via AppKit's setFrame animator. The menu-bar panel keeps this off
    /// because it has its own fade orchestration; the installer window flips
    /// it on so screen transitions resize smoothly instead of snapping.
    var animatesWindowResize: Bool = false

    /// When true, the post-resize frame is clamped to the host screen's
    /// `visibleFrame` so growth (which pins the top edge and pushes the
    /// bottom down) cannot slide the window under the Dock or above the
    /// menu bar. The menu-bar panel keeps this off because it's anchored
    /// to the menu bar by construction; floating windows that the user
    /// repositions (the installer is `window.center()`-ed at open) need
    /// the clamp because top-edge pinning otherwise compounds with growth
    /// to march the window off-screen on tall content / small displays.
    var clampsToScreen: Bool = false

    override var safeAreaInsets: NSEdgeInsets { NSEdgeInsets() }

    override func invalidateIntrinsicContentSize() {
        super.invalidateIntrinsicContentSize()
        syncWindowSize(animate: animatesWindowResize)
    }

    override func viewDidMoveToWindow() {
        super.viewDidMoveToWindow()
        // Initial sizing (window just opened) should never animate — there's
        // no prior frame to animate from.
        syncWindowSize(animate: false)
    }

    private func syncWindowSize(animate: Bool) {
        guard let window else { return }
        let size = intrinsicContentSize
        guard size.width > 0, size.height > 0 else { return }
        if window.frame.size == size { return }
        // Pin the top edge — both the menu-bar panel and the installer window
        // anchor visually at the top, so growth must expand downward.
        var frame = window.frame
        frame.origin.y += frame.size.height - size.height
        frame.size = size
        if clampsToScreen, let visibleFrame = (window.screen ?? NSScreen.main)?.visibleFrame {
            // Fix the top edge first, then the bottom — order matters so
            // the bottom adjustment wins when the window is taller than
            // the visible frame (degenerate; cap is supposed to prevent
            // it). Result in that case: top pinned to visibleFrame.minY,
            // bottom may extend above visibleFrame.maxY.
            if frame.maxY > visibleFrame.maxY {
                frame.origin.y = visibleFrame.maxY - frame.height
            }
            if frame.minY < visibleFrame.minY {
                frame.origin.y = visibleFrame.minY
            }
        }
        window.setFrame(frame, display: true, animate: animate)
    }
}
